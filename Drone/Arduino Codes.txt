1 :- UltraSonic

int tpin = 9;
int echopin = 10;
int led = 7;

void setup() {
  Serial.begin(9600);
  pinMode(led, OUTPUT);
  pinMode(tpin, OUTPUT);
  pinMode(echopin, INPUT);
}

void loop() {
  long duration, distance;
  digitalWrite(tpin, HIGH);
  delayMicroseconds(1000);
  digitalWrite(tpin, LOW);
  duration = pulseIn(echopin, HIGH);
  distance = (duration/2)/29.1;
  Serial.print(distance);
  Serial.println("CM");
  delay(10);

  if (distance <= 10) {
     digitalWrite(led, HIGH);
  }

  else if (distance > 10) {
     digitalWrite(led, LOW);
  }
}


2 :- LDR

const int ledpin = 13;
const int ldrpin = 5;

void setup() {
   Serial.begin(9600);
   pinMode(ledpin, OUTPUT);
   pinMode(ldrpin, INPUT);
}

void loop() {
  int ldrstatus = digitalRead(ldrpin);
  if (ldrstatus == 1) {
    digitalWrite(ledpin, HIGH);
    Serial.println(ldrstatus);
  }
  else {
    digitalWrite(ledpin, LOW);
    Serial.println(ldrstatus);
  }
}

3 :- Servo Motor

#include <Servo.h>
Servo s1;

void setup() {
  s1.attach(9);
}

void loop() {
  s1.write(0);
  delay(1000);
  s1.write(180);
  delay(1000);
  s1.write(0);
}
